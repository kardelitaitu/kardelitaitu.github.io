Modular Rust Architecture

// Example module composition in Main
use oz_pos::{StoreManager, InventoryModule, CloudSync};

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    // Load store and cashier configurations
    let store = StoreManager::init("./config/store.json").await?;
    
    // Mount modules dynamically
    store.register_module(InventoryModule::new())?;
    store.register_module(CloudSync::new(SyncInterval::Realtime))?;

    store.run().await?;
    Ok(())
}
