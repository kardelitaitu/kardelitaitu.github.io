White Label Customization via Lua

-- config.lua - Customize your brand & business rules
local config = {
  shop_name = "Premium Retailers",
  white_label = {
    brand_color = "#ea580c",
    receipt_header = "Welcome to Our Premium Store!",
    show_logo = true,
  },
  billing = {
    currency = "USD",
    tax_rate = 0.08, -- 8% sales tax
  },
  modules = {
    inventory = true,
    user_activity_logs = true, -- track detailed sessions
    live_analytics = true,     -- real-time dashboard data
    email_reports = {
      enabled = true,
      frequency = "daily",     -- "daily" | "weekly" | "monthly"
      recipients = {"owner@store.com", "manager@store.com"}
    },
    cloud_sync = {
      enabled = true,
      interval_seconds = 60,
    }
  }
}

-- Script custom dynamic business behaviors
function calculate_discount(cart)
  if cart.total > 100.00 then
    return cart.total * 0.10 -- 10% off orders > $100
  end
  return 0.0
end
